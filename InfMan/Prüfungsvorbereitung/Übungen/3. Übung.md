# 1 a)
```sql
create table Buch(
	ISBN char(13) primary key,
	Titel varchar(255),
	Preis decimal(10,2)
);

create table Auto(
	ID int primary key,
	Vorname varchar(80), 
	Nachname varchar(80)
);

create table BuchAutor(
	ISBN char(13),
	ID int,
	FOREIGN KEY (ISBN) REFERENCES Buch(ISBN),
	FOREIGN KEY (ID) REFERENCES Autor(ID)
);

create table Bibliothek(
	Name varchar(40) primary key,
	Ort varchar(80),
	Telefon varchar(40)
);

create table leser(
	ID int primary key,
	Vorname varchar(80),
	Nachname varchar(80)
);

create table Exemplar(
	ExemplarNr int, 
	ISBN char(13),
	Bibliothek varchar(40),
	Leser int,
	primary key (ExemplarNr, ISBN),
	foreign key (ISBN) references Buch(ISBN),
	foreign key (Bibliothek) references Bibliothek(Name),
	foreign key (Leser) references Leser(ID)
);
```
